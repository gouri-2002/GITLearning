git config --global user.name "gouri" [setting git user name]

git config --global user.email "gourios@gmail.com" [setting git user email]

git init [to intialize git in current working directory]

git status [list all untracked files]

git add -A [to add all untracked files to staging area]

git commit -m "message" [to commit with descriptive commit message]

git remote add remote_name path [adding remote]
eg: git remote add origin https://github.com/gouri-2002/GITLearning.git

git push remote_name branch []
eg: git push origin master

git branch [list branches]
